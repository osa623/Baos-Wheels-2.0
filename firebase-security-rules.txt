// Copy these rules to your Firebase console to fix the notification permissions issue

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow anyone to read public data
    match /community_chats/{document=**} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    match /community_replies/{document=**} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Fix for notifications - allow users to create and read their own notifications
    match /notifications/{notification} {
      // Allow users to read notifications addressed to them
      allow read: if request.auth != null && request.auth.uid == resource.data.userId;
      
      // Allow any authenticated user to create notifications
      allow create: if request.auth != null;
      
      // Only the recipient can mark notifications as read
      allow update: if request.auth != null && 
                     request.auth.uid == resource.data.userId && 
                     request.resource.data.diff(resource.data).affectedKeys().hasOnly(['isRead']);
      
      // Only admins can delete notifications
      allow delete: if false;
    }
    
    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
